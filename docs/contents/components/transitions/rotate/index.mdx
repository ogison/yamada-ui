## Import

```ts
import { Rotate } from "@yamada-ui/react"
```

## Usage

Pass the components to be switched as `from` and `to`.

```tsx functional=true
return <Rotate from={<Sun fontSize="6xl" />} to={<Moon fontSize="6xl" />} />
```

### Change the Rotation Angle

To change the rotation angle, set a value (in degrees) for `rotate`.

```tsx functional=true
return (
  <Rotate
    rotate={360}
    from={<Sun fontSize="6xl" />}
    to={<Moon fontSize="6xl" />}
  />
)
```

### Change the Duration

To change the duration, set a number (in seconds) to `duration`.

```tsx functional=true
return (
  <Rotate
    duration={0.6}
    from={<Sun fontSize="6xl" />}
    to={<Moon fontSize="6xl" />}
  />
)
```

### Delay the Animation Trigger

To delay the animation trigger, set a value (in seconds) for `delay`.

```tsx functional=true
return (
  <Rotate
    delay={1}
    from={<Sun fontSize="6xl" />}
    to={<Moon fontSize="6xl" />}
  />
)
```

### Disable

To disable, set `isDisabled` to `true`.

```tsx functional=true
return (
  <Rotate
    isDisabled
    from={<Sun fontSize="6xl" />}
    to={<Moon fontSize="6xl" />}
  />
)
```

### ReadOnly

To readyOnly, set `isReadOnly` to `true`.

```tsx functional=true
return (
  <Rotate
    isReadOnly
    from={<Sun fontSize="6xl" />}
    to={<Moon fontSize="6xl" />}
  />
)
```

### Control

```tsx functional=true
const [value, onChange] = useState<RotateIdent>("to")

return (
  <Rotate
    value={value}
    onChange={onChange}
    from={<Sun fontSize="6xl" />}
    to={<Moon fontSize="6xl" />}
  />
)
```
