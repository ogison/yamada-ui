## Import

```ts
import { Airy } from "@yamada-ui/react"
```

## Usage

Pass the components to be switched as `from` and `to`.

```tsx functional=true
return (
  <Airy
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### Change the Duration

To change the duration, set a number (in seconds) to `duration`.

```tsx functional=true
return (
  <Airy
    duration={0.6}
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### Delay the Animation Trigger

To delay the animation trigger, set a value (in seconds) for `delay`.

```tsx functional=true
return (
  <Airy
    delay={1}
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### Disable

To disable, set `isDisabled` to `true`.

```tsx functional=true
return (
  <Airy
    isDisabled
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### ReadOnly

To readyOnly, set `isReadOnly` to `true`.

```tsx functional=true
return (
  <Airy
    isReadOnly
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### Control

```tsx functional=true
const [value, onChange] = useState<AiryIdent>("to")

return (
  <Airy
    value={value}
    onChange={onChange}
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```
