## Theming

The `Airy` is a [single part component](/styled-system/theming/component-styles#single-part-components).

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

```ts
export const Airy: ComponentStyle<"Airy"> = {
  baseStyle: {
    userSelect: "none",
    width: "fit-content",
    _disabled: {
      cursor: "not-allowed",
      opacity: 0.4,
    },
    _readOnly: {
      cursor: "default",
    },
  },

  variants: {},

  sizes: {},

  defaultProps: {},
}
```
