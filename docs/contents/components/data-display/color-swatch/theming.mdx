## Theming

The `ColorSwatch` is a [multi part component](/styled-system/theming/component-styles#multi-part-components).

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

```ts
export const ColorSwatch: ComponentMultiStyle<"ColorSwatch"> = {
  baseStyle: {
    container: {},
    overlay: { boxSize: "full" },
  },

  sizes: {
    xs: {
      container: { boxSize: "6", rounded: "sm" },
      overlay: { rounded: "sm" },
    },
    sm: {
      container: { boxSize: "8", rounded: "md" },
      overlay: { rounded: "md" },
    },
    md: {
      container: { boxSize: "10", rounded: "md" },
      overlay: { rounded: "md" },
    },
    lg: {
      container: { boxSize: "12", rounded: "md" },
      overlay: { rounded: "md" },
    },
  },

  variants: {
    basic: {},
    rounded: {
      container: { rounded: "full" },
      overlay: { rounded: "full" },
    },
  },

  defaultProps: {
    size: "sm",
    variant: "basic",
  },
}
```
